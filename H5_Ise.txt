        // put your code here

    }
  
    public void wisselgeld()
    {
        int muntAantal;
        System.out.println("wisselgeld totaal: " + rest);
        while (rest > 0)

                muntAantal = rest / 100;

                {
                    System.out.println( muntAantal + " munten van 100 cent");

            }
            else if (rest <100 && rest >= 50)
            {

public class Prijsaanvraag
{
    // instance variables 
    private String naam;
    private String email;
    private int aantVolw;
    private int aantKind;
    private int aantNacht;

    /**
     * Constructor for objects of class Prijsaanvraag
     */
    public Prijsaanvraag(String naam, String email,int aantVolw, int aantKind, int aantNacht )
    {
        // initialise instance variables
        this.naam = naam;
        this.email = email;
        this.aantVolw = aantVolw;
        this.aantKind = aantKind;
        this.aantNacht = aantNacht;
    }

    /**
     * Calculate prices per group (aantVolw && aanKind) per night and the total amount of their stay.
     * Print out the message to the customer including their details and calculation.
     * 
     * @param   totaalVolw to calculate the price for aantVolw per night
     * @param   totaalKind to calculate the price for aantKind per night 
     * @param   totaal to calculate the total price for all visitors during their stay 
     * 
     */
    
    public void prijsopgave()
    {
        int totaalVolw;
        int totaalKind;
        int totaal;
        
        totaalVolw = aantVolw * 63;
        totaalKind = aantKind * 47;
        totaal = (totaalVolw + totaalKind) * aantNacht;
        
        System.out.println ("Geachte " + naam +"," + "\n" + "\n" +
            "Het totaalbedrag voor een verblijf in ons hotel met" + "\n" + "\n" +
            aantVolw + " volwassenen" + "\n" +
            aantKind + " kinderen" + "\n" + "\n" +
            "voor " + aantNacht + " overnachtingen bedraagt: " + totaal + " Euro." + "\n" + "\n" +
            "Wij bevestigen deze offerte per e-mail op het adres: " + email);
    }
}
