/**
 * class Teller: to add on or substract 1 point
 * 
 * @IseAn 
 * @21/12/2020
 */

public class Teller
{
    // instance variables - determin variable waarde
    private int waarde;

    // Constructor for objects of class Teller
    public Teller(int waarde)
    {
        // initialise instance variables
        waarde = this.waarde;
    }
    
    //to calculate the new score
    public void verhoog()
    {
        waarde += 1;
    }
    
    public void verlaag()
    {
        waarde -= 1 ;
    }
    //Getter
    
    public int getWaarde()
    {
        return waarde;
    }
    
    public void setWaarde(int waarde)
    {
        this.waarde = waarde;
    }
     
    //visualize the Teller
    public String toString()
      {
        return waarde + "";
       }
}

public class Scorebord
{
    // instance variables - call Teller and name 2 variables for Scorebord
    private Teller ploeg1;
    private Teller ploeg2;

    {
        // initialise instance variables

    }

    //visualize the score
    public String toString()
    {
        // put your code here
        return ploeg1 + "-" + ploeg2;
    }
    
    //print the score
    public void print()
    {
        System.out.println("De score is: " + ploeg1 + "-" + ploeg2);
    }
}
