/**
 * To add and substract articles in class Voorraad
 * 
 * @author Ise An Verstegen 
 * @version 1/5/2021
 * 
 */
public class Voorraad
{
    //instances inherated to Boeken and Tijdschriften
    protected String titel;
    protected double prijs;
    protected int aantal;

    /**
     * Methods:
     * 
     * to get the amount of articles per article (getAantal())
     * to get the price of an article (getPrice())
     * to add articles (aanvullen)
     * to substract articles after sale (verkoop())
     * to print the data of an article (print())
     */
    public int getAantal()
    {
        return aantal;
    }
    
    public double getPrijs()
    {
        return prijs;
    }
    
    public void aanvullen(int nieuwAantal)
    {
        aantal += nieuwAantal;
    }
    
    public void verkoop(int nieuwAantal)
    {
        aantal -= nieuwAantal;
    }
  
    public void print() 
    {
        System.out.println( "Titel: " + titel );
        System.out.printf( "Prijs: %.2f%n", prijs );
        System.out.println("Aantal: " + aantal);
    }
}


/**
 * With the purpose to display the stock of the bookstore class Boekenwinkel
 * 
 * @author Ise An Verstegen 
 * @version 1/5/2021
 * 
 */

import java.util.*;

public class Boekhandel
{
    //@param lijst to make a list of class Voorrraad (stock)
    //@param totaalPrijs to calculate the total price of the articles in stock
    
    private ArrayList<Voorraad> lijst;
    private double totaalPrijs;
   
    /**
     * Constructor for objects of class Overzicht
     * @param lijst to make a list of class Voorrraad (stock)
     * 
     */
    public Boekhandel()
    {
        // initialise instance variables
        lijst = new ArrayList<Voorraad>();
    }

    /**
     * to add articles to the list of Voorraad (voegArtikelToe())
     * to print the overview of both stock and total amount of stock of class Voorraad(printOverzicht()) 
     * 
     * @param  totaalPrijsArtikel to calculate the total price of the number of articles present in stock
     *
     */
    public void voegArtikelToe(Voorraad v)
    {
        lijst.add(v);
    }
    
    public void totaleVerkoopprijsVoorraad() {
        double totaalPrijsArtikel;
        for (Voorraad v : lijst)
            {
                totaalPrijsArtikel = v.getPrijs() * v.getAantal();
                totaalPrijs += totaalPrijsArtikel;
            }  
        }
    
    public void printOverzicht()
    {
        for(Voorraad v : lijst)
        {
            v.print();
        }
        System.out.printf("Totale Verkoopprijs Voorraad: %.2f%n", totaalPrijs);
        System.out.println("-----------------------");
    }
}
